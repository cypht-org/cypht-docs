110: trait Hm_List {

116:      * Add a message
117:      * @param string $string message to add
118:      * @return void
119:      */
120:     public static function add($string) {
121:         self::$msgs[] = self::str($string, false);
122:     }
123: 
124:     /**
125:      * Return all messages
126:      * @return array all messages
127:      */
128:     public static function get() {
129:         return self::$msgs;
130:     }
131: 
132:     /**
133:      * Stringify a value
134:      * @param mixed $mixed value to stringify
135:      * @return string
136:      */
137:     public static function str($mixed, $return_type=true) {
138:         $type = gettype($mixed);
139:         if (in_array($type, array('array', 'object'), true)) {
140:             $str = print_r($mixed, true);
141:         }
142:         elseif ($return_type) {
143:             $str = sprintf("%s: %s", $type, $mixed);
144:         }
145:         else {
146:             $str = (string) $mixed;
147:         }
148:         return $str;
149:     }
150: 
151:     /**
152:      * Show all messages
153:      * @param string $type can be one of "print", "log", or "return"
154:      * @return mixed
155:      */
156:     public static function show($type='print') {
157:         if ($type == 'log') {
158:             Hm_Functions::error_log(print_r(self::$msgs, true));
159:         }
160:         elseif ($type == 'return') {
161:             return self::$msgs;
162:         }
163:         else {
164:             print_r(self::$msgs);
165:         }
166:     }
