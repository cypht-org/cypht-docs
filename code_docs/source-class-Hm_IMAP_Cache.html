 102:     /**
 103:      * break up a "line" response from imap. If we find
 104:      * a literal we read ahead on the stream and include it.
 105:      *
 106:      * @param string $line data read from the IMAP server
 107:      * @param int $current_size size of current read operation
 108:      * @param int $max maximum input size to allow
 109:      * @param int $line_length chunk size to read literals with
 110:      *
 111:      * @return array a line continuation marker and the parsed data
 112:      *               from the IMAP server
 113:      */
 114:     protected function parse_line($line, $current_size, $max, $line_length) {
 115:         /* make it a bit easier to find "atoms" */
 116:         $line = str_replace(')(', ') (', $line);
 117:         $this->literal_overflow = false;
 118: 
 119:         /* will hold the line parts */
 120:         $parts = array();
 121: 
 122:         /* flag to control if the line continues */
 123:         $line_cont = false;
 124: 
 125:         /* line size */
 126:         $len = strlen($line);
 127: 
 128:         /* walk through the line */
 129:         for ($i=0;$i<$len;$i++) {
 130: 
 131:             /* this will hold one "atom" from the parsed line */
 132:             $chunk = '';
 133: 
 134:             /* if we hit a newline exit the loop */
 135:             if ($line{$i} == "\r" || $line{$i} == "\n") {
 136:                 $line_cont = false;
 137:                 break;
 138:             }
 139: 
 140:             /* skip spaces */
 141:             if ($line{$i} == ' ') {
 142:                 continue;
 143:             }
 144: 
 145:             /* capture special chars as "atoms" */
 146:             elseif ($line{$i} == '*' || $line{$i} == '[' || $line{$i} == ']' || $line{$i} == '(' || $line{$i} == ')') {
 147:                 $chunk = $line{$i};
 148:             }
 149:         
 150:             /* regex match a quoted string */
 151:             elseif ($line{$i} == '"') {
 152:                 if (preg_match("/^(\"[^\"\\\]*(?:\\\.[^\"\\\]*)*\")/", substr($line, $i), $matches)) {
 153:                     $chunk = substr($matches[1], 1, -1);
 154:                 }
 155:                 $i += strlen($chunk) + 1;
 156:             }
 157: 
 158:             /* IMAP literal */
 159:             elseif ($line{$i} == '{') {
 160:                 $end = strpos($line, '}');
 161:                 if ($end !== false) {
 162:                     $literal_size  = substr($line, ($i + 1), ($end - $i - 1));
 163:                 }
 164:                 $lit_result = $this->read_literal($literal_size, $max, $current_size, $line_length);
 165:                 $chunk = $lit_result[0];
 166:                 if (!isset($lit_result[1]) || $lit_result[1] != "\r\n") {
 167:                     $line_cont = true;
 168:                 }
 169:                 if (isset($lit_result[1]) && $lit_result[1] != "\r\n" && strlen($lit_result[1]) > 0) {
 170:                     $this->literal_overflow = $lit_result[1];
 171:                 }
 172:                 $i = $len;
 173:             }
 174: 
 175:             /* all other atoms */
 176:             else {
 177:                 $marker = -1;
 178: 
 179:                 /* don't include these three trailing chars in the atom */
 180:                 foreach (array(' ', ')', ']') as $v) {
 181:                     $tmp_marker = strpos($line, $v, $i);
 182:                     if ($tmp_marker !== false && ($marker == -1 || $tmp_marker < $marker)) {
 183:                         $marker = $tmp_marker;
 184:                     }
 185:                 }
 186: 
 187:                 /* slice out the chunk */
 188:                 if ($marker !== false && $marker !== -1) {
 189:                     $chunk = substr($line, $i, ($marker - $i));
 190:                     $i += strlen($chunk) - 1;
 191:                 }
 192:                 else {
 193:                     $chunk = rtrim(substr($line, $i));
 194:                     $i += strlen($chunk);
 195:                 }
 196:             }

 198:             /* if we found a worthwhile chunk add it to the results set */
 199:             if ($chunk) {
 200:                 $parts[] = $chunk;
 201:             }
 202:         }
 203:         return array($line_cont, $parts);

 225:     /**
 226:      * loop through "lines" returned from imap and parse them with parse_line() and read_literal.
 227:      * it can return the lines in a raw format, or parsed into atoms. It also supports a maximum
 228:      * number of lines to return, in case we did something stupid like list a loaded unix homedir
 229:      *
 230:      * @param int $max max size of response allowed
 231:      * @param bool $chunked flag to parse the data into IMAP "atoms"
 232:      * @param int $line_length chunk size to read in literals using fgets
 233:      * @param bool $sort flag for non-compliant sort result parsing speed up
 234:      *
 235:      * @return array of parsed or raw results
 236:      */
 237:     protected function get_response($max=false, $chunked=false, $line_length=8192, $sort=false) {
 238:         /* defaults and results containers */
 239:         $result = array();
 240:         $current_size = 0;
 241:         $chunked_result = array();
 242:         $last_line_cont = false;
 243:         $line_cont = false;
 244:         $c = -1;
 245:         $n = -1;
 246: 
 247:         /* start of do -> while loop to read from the IMAP server */
 248:         do {
 249:             $n++;
 250: 
 251:             /* if we loose connection to the server while reading terminate */
 252:             if (!is_resource($this->handle) || feof($this->handle)) {
 253:                 break;
 254:             }
 255: 
 256:             /* read in a line up to 8192 bytes */
 257:             $result[$n] = $this->fgets($line_length);
 258: 
 259:             /* keep track of how much we have read and break out if we max out. This can
 260:              * happen on large messages. We need this check to ensure we don't exhaust available
 261:              * memory */
 262:             $current_size += strlen($result[$n]);
 263:             if ($max && $current_size > $max) {
 264:                 $this->max_read = true;
 265:                 break;
 266:             }
 267: 
 268:             /* if the line is longer than 8192 bytes keep appending more reads until we find
 269:              * an end of line char. Keep checking the max read length as we go */
 270:             while(substr($result[$n], -2) != "\r\n" && substr($result[$n], -1) != "\n") {
 271:                 if (!is_resource($this->handle) || feof($this->handle)) {
 272:                     break;
 273:                 }
 274:                 $result[$n] .= $this->fgets($line_length);
 275:                 if ($result[$n] === false) {
 276:                     break;
 277:                 }
 278:                 $current_size += strlen($result[$n]);
 279:                 if ($max && $current_size > $max) {
 280:                     $this->max_read = true;
 281:                     break 2;
 282:                 }
 283:             }
 284: 
 285:             /* check line continuation marker and grab previous index and parsed chunks */
 286:             if ($line_cont) {
 287:                 $last_line_cont = true;
 288:                 $pres = $n - 1;
 289:                 if ($chunks) {
 290:                     $pchunk = $c;
 291:                 }
 292:             }
 293: 
 294:             /* If we are using quick parsing of the IMAP SORT response we know the results are simply space
 295:              * delimited UIDs so quickly explode(). Otherwise we have to follow the spec and look for quoted
 296:              * strings and literals in the parse_line() routine. */
 297:             if ($sort) {
 298:                 $line_cont = false;
 299:                 $chunks = explode(' ', trim($result[$n]));
 300:             }
 301: 
 302:             /* properly parse the line */
 303:             else {
 304:                 list($line_cont, $chunks) = $this->parse_line($result[$n], $current_size, $max, $line_length);
 305:                 while ($this->literal_overflow) {
 306:                     $lit_text = $this->literal_overflow;
 307:                     $this->literal_overflow = false;
 308:                     $current_size += strlen($lit_text);
 309:                     list($line_cont, $new_chunks) = $this->parse_line($lit_text, $current_size, $max, $line_length);
 310:                     $chunks = array_merge($chunks, $new_chunks);
 311:                 }
 312:             }
 313: 
 314:             /* merge lines that should have been recieved as one and add to results */
 315:             if ($chunks && !$last_line_cont) {
 316:                 $c++;
 317:             }
 318:             if ($last_line_cont) {
 319:                 $result[$pres] .= ' '.implode(' ', $chunks);
 320:                 if ($chunks) {
 321:                     $line_bits = array_merge($chunked_result[$pchunk], $chunks);
 322:                     $chunked_result[$pchunk] = $line_bits;
 323:                 }
 324:                 $last_line_cont = false;
 325:             }
 326: 
 327:             /* add line and parsed bits to result set */
 328:             else {
 329:                 $result[$n] = implode(' ', $chunks);
 330:                 if ($chunked) {
 331:                     $chunked_result[$c] = $chunks;
 332:                 }
 333:             }
 334: 
 335:             /* check for untagged error condition. This represents a server problem but there is no reason
 336:              * we can't attempt to recover with the partial response we received up until this point */
 337:             if (substr(strtoupper($result[$n]), 0, 6) == '* BYE ') {
 338:                 break;
 339:             }
 340: 
 341:         /* end outer loop when we receive the tagged response line */
 342:         } while (substr($result[$n], 0, strlen('A'.$this->command_count)) != 'A'.$this->command_count);
 343: 
 344:         /* return either raw or parsed result */
 345:         $this->responses[] = $result;
 346:         if ($chunked) {
 347:             $result = $chunked_result;
 348:         }
 349:         if ($this->current_command && isset($this->commands[$this->current_command])) {
 350:             $start_time = $this->commands[$this->current_command];
 351:             $this->commands[$this->current_command] = microtime(true) - $start_time;
 352:             if (count($this->commands) >= $this->max_history) {
 353:                 array_shift($this->commands);
 354:                 array_shift($this->responses);
 355:             }
 356:         }
 357:         return $result;
 358:     }
