 57:     /**
 58:      * Process request details
 59:      * @param array $filters list of input filters from module sets
 60:      * @return void
 61:      */
 62:     public function __construct($filters) {
 63:         $this->filter_request_input($filters);
 64:         $this->get_other_request_details($filters);
 65:         $this->empty_super_globals();
 66: 
 67:         Hm_Debug::add('Using sapi: '.$this->sapi);
 68:         Hm_Debug::add('Request type: '.$this->type);
 69:         Hm_Debug::add('Request path: '.$this->path);
 70:         Hm_Debug::add('TLS request: '.intval($this->tls));
 71:         Hm_Debug::add('Mobile request: '.intval($this->mobile));
 72:     }

 74:     /**
 75:      * Sanitize and filter user and server input
 76:      * @param array $filters list of input filters from module sets
 77:      * @return void
 78:      */
 79:     private function filter_request_input($filters) {
 80:         if (array_key_exists('allowed_server', $filters)) {
 81:             $this->server = $this->filter_input(INPUT_SERVER, $_SERVER, $filters['allowed_server']);
 82:         }
 83:         if (array_key_exists('allowed_post', $filters)) {
 84:             $this->post = $this->filter_input(INPUT_POST, $_POST, $filters['allowed_post']);
 85:         }
 86:         if (array_key_exists('allowed_get', $filters)) {
 87:             $this->get = $this->filter_input(INPUT_GET, $_GET, $filters['allowed_get']);
 88:         }
 89:         if (array_key_exists('allowed_cookie', $filters)) {
 90:             $this->cookie = $this->filter_input(INPUT_COOKIE, $_COOKIE, $filters['allowed_cookie']);
 91:         }
 92:     }

 94:     /**
 95:      * Collect other useful details about a request
 96:      * @param array $filters list of input filters from module sets
 97:      * @return void
 98:      */
 99:     private function get_other_request_details($filters) {
100:         $this->sapi = php_sapi_name();
101:         if (array_key_exists('allowed_output', $filters)) {
102:             $this->allowed_output = $filters['allowed_output'];
103:         }
104:         if (array_key_exists('REQUEST_URI', $this->server)) {
105:             $this->path = $this->get_clean_url_path($this->server['REQUEST_URI']);
106:         }
107:         $this->get_request_type();
108:         $this->is_tls();
109:         $this->is_mobile();
110:     }

114:      * @return void
115:      */
116:     private function empty_super_globals() {
117:         $_POST = array();
118:         $_SERVER = array();
119:         $_GET = array();
120:         $_COOKIE = array();
121:         $_FILES = array();
122:         $_REQUEST = array();
123:         $_ENV = array();
124:         $GLOBALS = array('_SERVER' => array(), '_POST' => array(), '_GET' => array(), '_COOKIE' => array(),
125:             '_FILES' => array(), '_REQUEST' => array(), '_ENV' => array());
126:     }

128:     /**
129:      * Filter specified input against module defined filters
130:      * @param type string the type of input (POST, GET, COOKIE, etc)
131:      * @param filters array list of input filters from module sets
132:      * @return array filtered input data
133:      */
134:     private function filter_input($type, $data, $filters) {
135:         $data = filter_var_array($data, $filters, false);
136:         if (!$data) {
137:             return array();
138:         }
139:         if ($type == INPUT_GET) {
140:             $this->invalid_input_detected = count($_GET) > count($data);
141:             if ($this->invalid_input_detected) {
142:                 $this->invalid_input_fields = array_keys(array_diff_assoc($_GET, $data));
143:             }
144:         }
145:         if ($type == INPUT_POST) {
146:             $this->invalid_input_detected = count($_POST) > count($data);
147:             if ($this->invalid_input_detected) {
148:                 $this->invalid_input_fields = array_keys(array_diff_assoc($_POST, $data));
149:             }
150:         }
151:         return $data;
152:     }

154:     /**
155:      * Look at the HTTP_USER_AGENT value and set a mobile OS flag
156:      * @return void
157:      */
158:     private function is_mobile() {
159:         if (array_key_exists('HTTP_USER_AGENT', $this->server)) {
160:             if (preg_match("/(iphone|ipod|ipad|android|blackberry|webos)/i", $this->server['HTTP_USER_AGENT'])) {
161:                 $this->mobile = true;
162:             }
163:         }
164:     }

166:     /**
167:      * Determine if a request was done over TLS
168:      * @return void
169:      */
170:     private function is_tls() {
171:         if (array_key_exists('HTTPS', $this->server) && strtolower($this->server['HTTPS']) == 'on') {
172:             $this->tls = true;
173:         }
174:         elseif (array_key_exists('REQUEST_SCHEME', $this->server) && strtolower($this->server['REQUEST_SCHEME']) == 'https') {
175:             $this->tls = true;
176:         }
177:     }

179:     /**
180:      * Determine the request type, either AJAX or HTTP
181:      * @return void
182:      */
183:     private function get_request_type() {
184:         if ($this->is_ajax()) {
185:             $this->type = 'AJAX';
186:             $this->format = 'Hm_Format_JSON';
187:         }
188:         else {
189:             $this->type = 'HTTP';
190:             $this->format = 'Hm_Format_HTML5';
191:         }
192:     }

194:     /**
195:      * Determine if a request is an AJAX call
196:      * @return bool true if the request is from an AJAX call
197:      */
198:     private function is_ajax() {
199:         return array_key_exists('HTTP_X_REQUESTED_WITH', $this->server) && strtolower($this->server['HTTP_X_REQUESTED_WITH']) === 'xmlhttprequest';
200:     }

202:     /**
203:      * Make sure a url path is sane
204:      * @param string $uri path to check
205:      * @return string clean url path
206:      */
207:     private function get_clean_url_path($uri) {
208:         if (strpos($uri, '?') !== false) {
209:             $parts = explode('?', $uri, 2);
210:             $path = $parts[0];
211:         }
212:         else {
213:             $path = $uri;
214:         }
215:         if (substr($path, -1) != '/') {
216:             $path .= '/';
217:         }
218:         return $path;
219:     }
