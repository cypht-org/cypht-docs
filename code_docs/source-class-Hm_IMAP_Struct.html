 142:                 $chunk = $line{$i};
 143:             }
 144:         
 145:             /* regex match a quoted string */
 146:             elseif ($line{$i} == '"') {
 147:                 if (preg_match("/^(\"[^\"\\\]*(?:\\\.[^\"\\\]*)*\")/", substr($line, $i), $matches)) {
 148:                     $chunk = substr($matches[1], 1, -1);
 149:                 }
 150:                 $i += strlen($chunk) + 1;
 151:             }
 152: 
 153:             /* IMAP literal */
 154:             elseif ($line{$i} == '{') {
 155:                 $end = strpos($line, '}');
 156:                 if ($end !== false) {
 157:                     $literal_size  = substr($line, ($i + 1), ($end - $i - 1));
 158:                 }
 159:                 $lit_result = $this->read_literal($literal_size, $max, $current_size, $line_length);
 160:                 $chunk = $lit_result[0];
 161:                 if (!isset($lit_result[1]) || $lit_result[1] != "\r\n") {
 162:                     $line_cont = true;
 163:                 }
 164:                 if (isset($lit_result[1]) && $lit_result[1] != "\r\n" && strlen($lit_result[1]) > 0) {
 165:                     $this->literal_overflow = $lit_result[1];
 166:                 }
 167:                 $i = $len;
 168:             }
 169: 
 170:             /* all other atoms */
 171:             else {
 172:                 $marker = -1;
 173: 
 174:                 /* don't include these three trailing chars in the atom */
 175:                 foreach (array(' ', ')', ']') as $v) {
 176:                     $tmp_marker = strpos($line, $v, $i);
 177:                     if ($tmp_marker !== false && ($marker == -1 || $tmp_marker < $marker)) {
 178:                         $marker = $tmp_marker;
 179:                     }
 180:                 }
 181: 
 182:                 /* slice out the chunk */
 183:                 if ($marker !== false && $marker !== -1) {
 184:                     $chunk = substr($line, $i, ($marker - $i));
 185:                     $i += strlen($chunk) - 1;
 186:                 }
 187:                 else {
 188:                     $chunk = rtrim(substr($line, $i));
 189:                     $i += strlen($chunk);
 190:                 }
 191:             }
 192: 
 193:             /* if we found a worthwhile chunk add it to the results set */
 194:             if ($chunk) {
 195:                 $parts[] = $chunk;
 196:             }
 197:         }
 198:         return array($line_cont, $parts);
 199:     }

 201:     /**
 202:      * wrapper around fgets using $this->handle
 203:      * @param int $len max read length for fgets
 204:      * @return string data read from the IMAP server
 205:      */
 206:     protected function fgets($len=false) {
 207:         if (is_resource($this->handle) && !feof($this->handle)) {
 208:             if ($len) {
 209:                 return fgets($this->handle, $len);
 210:             }
 211:             else {
 212:                 return fgets($this->handle);
 213:             }
 214:         }
 215:         return '';
 216:     }

 218:     /**
 219:      * loop through "lines" returned from imap and parse them with parse_line() and read_literal.
 220:      * it can return the lines in a raw format, or parsed into atoms. It also supports a maximum
 221:      * number of lines to return, in case we did something stupid like list a loaded unix homedir
 222:      * @param int $max max size of response allowed
 223:      * @param bool $chunked flag to parse the data into IMAP "atoms"
 224:      * @param int $line_length chunk size to read in literals using fgets
 225:      * @param bool $sort flag for non-compliant sort result parsing speed up
 226:      * @return array of parsed or raw results
 227:      */
 228:     protected function get_response($max=false, $chunked=false, $line_length=8192, $sort=false) {
 229:         /* defaults and results containers */
 230:         $result = array();
 231:         $current_size = 0;
 232:         $chunked_result = array();
 233:         $last_line_cont = false;
 234:         $line_cont = false;
 235:         $c = -1;
 236:         $n = -1;
 237: 
 238:         /* start of do -> while loop to read from the IMAP server */
 239:         do {
 240:             $n++;
 241: 
 242:             /* if we loose connection to the server while reading terminate */
 243:             if (!is_resource($this->handle) || feof($this->handle)) {
 244:                 break;
 245:             }
 246: 
 247:             /* read in a line up to 8192 bytes */
 248:             $result[$n] = $this->fgets($line_length);
 249: 
 250:             /* keep track of how much we have read and break out if we max out. This can
 251:              * happen on large messages. We need this check to ensure we don't exhaust available
 252:              * memory */
 253:             $current_size += strlen($result[$n]);
 254:             if ($max && $current_size > $max) {
 255:                 $this->max_read = true;
 256:                 break;
 257:             }
 258: 
 259:             /* if the line is longer than 8192 bytes keep appending more reads until we find
 260:              * an end of line char. Keep checking the max read length as we go */
 261:             while(substr($result[$n], -2) != "\r\n" && substr($result[$n], -1) != "\n") {
 262:                 if (!is_resource($this->handle) || feof($this->handle)) {
 263:                     break;
 264:                 }
 265:                 $result[$n] .= $this->fgets($line_length);
 266:                 if ($result[$n] === false) {
 267:                     break;
 268:                 }
 269:                 $current_size += strlen($result[$n]);
 270:                 if ($max && $current_size > $max) {
 271:                     $this->max_read = true;
 272:                     break 2;
 273:                 }
 274:             }
 275: 
 276:             /* check line continuation marker and grab previous index and parsed chunks */
 277:             if ($line_cont) {
 278:                 $last_line_cont = true;
 279:                 $pres = $n - 1;
 280:                 if ($chunks) {
 281:                     $pchunk = $c;
 282:                 }
 283:             }
 284: 
 285:             /* If we are using quick parsing of the IMAP SORT response we know the results are simply space
 286:              * delimited UIDs so quickly explode(). Otherwise we have to follow the spec and look for quoted
 287:              * strings and literals in the parse_line() routine. */
 288:             if ($sort) {
 289:                 $line_cont = false;
 290:                 $chunks = explode(' ', trim($result[$n]));
 291:             }
 292: 
 293:             /* properly parse the line */
 294:             else {
 295:                 list($line_cont, $chunks) = $this->parse_line($result[$n], $current_size, $max, $line_length);
 296:                 while ($this->literal_overflow) {
 297:                     $lit_text = $this->literal_overflow;
 298:                     $this->literal_overflow = false;
 299:                     $current_size += strlen($lit_text);
 300:                     list($line_cont, $new_chunks) = $this->parse_line($lit_text, $current_size, $max, $line_length);
 301:                     $chunks = array_merge($chunks, $new_chunks);
 302:                 }
 303:             }
 304: 
 305:             /* merge lines that should have been recieved as one and add to results */
 306:             if ($chunks && !$last_line_cont) {
 307:                 $c++;
 308:             }
 309:             if ($last_line_cont) {
 310:                 $result[$pres] .= ' '.implode(' ', $chunks);
 311:                 if ($chunks) {
 312:                     $line_bits = array_merge($chunked_result[$pchunk], $chunks);
 313:                     $chunked_result[$pchunk] = $line_bits;
 314:                 }
 315:                 $last_line_cont = false;
 316:             }
 317: 
 318:             /* add line and parsed bits to result set */
 319:             else {
 320:                 $result[$n] = implode(' ', $chunks);
 321:                 if ($chunked) {
 322:                     $chunked_result[$c] = $chunks;
 323:                 }
 324:             }
 325: 
 326:             /* check for untagged error condition. This represents a server problem but there is no reason
 327:              * we can't attempt to recover with the partial response we received up until this point */
 328:             if (substr(strtoupper($result[$n]), 0, 6) == '* BYE ') {
 329:                 break;
 330:             }
 331: 
 332:             /* check for challenge strings */
 333:             if (substr($result[$n], 0, 1) == '+') {
 334:                 if (preg_match("/^\+ ([a-zA-Z0-9=]+)$/", $result[$n], $matches)) {
 335:                     break;
 336:                 }
 337:             }
 338: 
 339:         /* end outer loop when we receive the tagged response line */
 340:         } while (substr($result[$n], 0, strlen('A'.$this->command_count)) != 'A'.$this->command_count);
 341: 
 342:         /* return either raw or parsed result */
 343:         $this->responses[] = $result;
 344:         if ($chunked) {
 345:             $result = $chunked_result;
 346:         }
 347:         if ($this->current_command && isset($this->commands[$this->current_command])) {
 348:             $start_time = $this->commands[$this->current_command];
 349:             $this->commands[$this->current_command] = microtime(true) - $start_time;
 350:             if (count($this->commands) >= $this->max_history) {
 351:                 array_shift($this->commands);
 352:                 array_shift($this->responses);
 353:             }
 354:         }
 355:         return $result;
 356:     }

 358:     /**
 359:      * put a prefix on a command and send it to the server
 360:      * @param mixed $command IMAP command
 361:      * @param bool $no_prefix flag to skip adding the prefix
 362:      * @return void
 363:      */
 364:     protected function send_command($command, $no_prefix=false) {
 365:         $this->cached_response = false;
 366:         if (!$no_prefix) {
 367:             $command = 'A'.$this->command_number().' '.$command;
 368:         }
 369: 
 370:         /* send the command out to the server */
 371:         if (is_resource($this->handle)) {
 372:             $res = @fputs($this->handle, $command);
 373:             if (!$res) {
 374:                 $this->debug[] = 'Error communicating with IMAP server: '.trim($command);
 375:             }
 376:         }
 377: 
 378:         /* save the command and time for the IMAP debug output option */
 379:         if (strstr($command, 'LOGIN')) {
 380:             $command = 'LOGIN';
 381:         }
 382:         $this->commands[trim($command)] = microtime( true );
 383:         $this->current_command = trim($command);
 384:     }

 389:      * @param bool $chunked flag defining the type of $data
 390:      * @return bool true to indicate a success response from the IMAP server
 391:      */
 392:     protected function check_response($data, $chunked=false, $log_failures=true) {
 393:         $result = false;
 394: 
 395:         /* find the last bit of the parsed response and look for the OK atom */
 396:         if ($chunked) {
 397:             if (!empty($data) && isset($data[(count($data) - 1)])) {
 398:                 $vals = $data[(count($data) - 1)];

 411:                 $result = true;
 412:             }
 413:         }
 414:         if (!$result && $log_failures) {
 415:             $this->debug[] = 'Command FAILED: '.$this->current_command;
 416:         }
 417:         return $result;
 418:     }

 444:     /**
 445:      * type checks
 446:      * @param string $val value to check
 447:      * @param string $type type of value to check against
 448:      * @return bool true if the type check passed
 449:      */
 450:     protected function input_validate($val, $type) {
 451:         $imap_search_charsets = array(
 452:             'UTF-8',
 453:             'US-ASCII',
 454:             '',
 455:         );
 456:         $imap_keywords = array(
 457:             'ARRIVAL',    'DATE',    'FROM',      'SUBJECT',
 458:             'CC',         'TO',      'SIZE',      'UNSEEN',
 459:             'SEEN',       'FLAGGED', 'UNFLAGGED', 'ANSWERED',
 460:             'UNANSWERED', 'DELETED', 'UNDELETED', 'TEXT',
 461:             'ALL', 'DRAFT', 'NEW', 'RECENT', 'OLD', 'UNDRAFT'
 462:         );
 463:         $valid = false;
 464:         switch ($type) {
 465:             case 'search_str':
 466:                 if (preg_match("/^[^\r\n]+$/", $val)) {
 467:                     $valid = true;
 468:                 }
 469:                 break;
 470:             case 'msg_part':
 471:                 if (preg_match("/^[\d\.]+$/", $val)) {
 472:                     $valid = true;
 473:                 }
 474:                 break;
 475:             case 'charset':
 476:                 if (!$val || in_array(strtoupper($val), $imap_search_charsets)) {
 477:                     $valid = true;
 478:                 }
 479:                 break;
 480:             case 'uid':
 481:                 if (ctype_digit((string) $val)) {
 482:                     $valid = true;
 483:                 }
 484:                 break;
 485:             case 'uid_list';
 486:                 if (preg_match("/^(\d+\s*,*\s*|(\d+|\*):(\d+|\*))+$/", $val)) {
 487:                     $valid = true;
 488:                 }
 489:                 break;
 490:             case 'mailbox';
 491:                 if (preg_match("/^[^\r\n]+$/", $val)) {
 492:                     $valid = true;
 493:                 }
 494:                 break;
 495:             case 'keyword';
 496:                 if (in_array(strtoupper($val), $imap_keywords)) {
 497:                     $valid = true;
 498:                 }
 499:                 break;
 500:         }
 501:         return $valid;
 502:     }
