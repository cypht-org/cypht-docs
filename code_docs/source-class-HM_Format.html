11: /**
12:  * Base class for output formatting. Currently JSON and HTML5 formats are
13:  * supported. To add support for a new format this class must be extended
14:  * and the content method needs to be overridden.
15:  * @abstract
16:  */
17: abstract class HM_Format {
18: 
19:     /**
20:      * Return combined output from all modules. Must be overridden by specific
21:      * output classes
22:      * @param array $output data from the output modules
23:      * @param array $allowed_output allowed fields for JSON responses
24:      * @return mixed combined output
25:      */
26:     abstract protected function content($input, $allowed_output);
27: }
28: 
29: /**
30:  * Handles JSON formatted results for AJAX requests
31:  */
32: class Hm_Format_JSON extends HM_Format {
33: 
34:     /**
35:      * Run modules and merge + filter the result array
36:      * @param array $input data from the handler modules
37:      * @param array $lang_str langauge strings
38:      * @param array $allowed_output allowed fields for JSON responses
39:      * @return JSON encoded data to be sent to the browser
40:      */
41:     public function content($output, $allowed_output) {
42:         $output['router_user_msgs'] = Hm_Msgs::get();
43:         $output = $this->filter_output($output, $allowed_output);
44:         return json_encode($output, JSON_FORCE_OBJECT);
45:     }
46: 
47:     /**
48:      * Filter data against module set white lists before sending it to the browser
49:      * @param array $data output module data to filter
50:      * @param array $allowed set of white list filters
51:      * @return array filtered data
52:      */
53:     public function filter_output($data, $allowed) {
54:         foreach ($data as $name => $value) {
55:             if (!array_key_exists($name, $allowed)) {
56:                 unset($data[$name]);
57:             }
58:             else {
59:                 if ($allowed[$name][1]) {
60:                     $new_value = filter_var($value, $allowed[$name][0], $allowed[$name][1]);
61:                 }
62:                 else {
63:                     $new_value = filter_var($value, $allowed[$name][0]);
64:                 }
65:                 if ($new_value === false && $allowed[$name] != FILTER_VALIDATE_BOOLEAN) {
66:                     unset($data[$name]);
67:                 }
68:                 else {
69:                     $data[$name] = $new_value;
70:                 }
71:             }
72:         }
73:         return $data;
74:     }
75: }
76: 
77: /**
78:  * Handles HTML5 formatted results for normal HTTP requests
79:  */
80: class Hm_Format_HTML5 extends HM_Format {
81: 
82:     /**
83:      * Collect and return content from modules for HTTP requests
84:      * @param array $output data from the output modules
85:      * @param array $allowed_output allowed fields for JSON responses
86:      * @return string HTML5 content
87:      */
88:     public function content($output, $allowed_output) {
89:         return implode('', $output);
90:     }
91: }
