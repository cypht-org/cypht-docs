 54: /**
 55:  * Build a human readable interval string
 56:  * @subpackage core/functions
 57:  * @param string $date_str date string parsable by strtotime()
 58:  * @return string
 59:  */
 60: function human_readable_interval($date_str) {
 61:     $precision     = 2;
 62:     $interval_time = array();
 63:     $now           = time();
 64:     $date          = strtotime($date_str);
 65:     $interval      = $now - $date;
 66:     $res           = array();
 67: 
 68:     $t['second'] = 1;
 69:     $t['minute'] = $t['second']*60;
 70:     $t['hour']   = $t['minute']*60;
 71:     $t['day']    = $t['hour']*24;
 72:     $t['week']   = $t['day']*7;
 73:     $t['month']  = $t['day']*30;
 74:     $t['year']   = $t['week']*52;
 75: 
 76:     if ($interval < 0) {
 77:         $interval += $t['hour'];
 78:         if ($interval < 0) {
 79:             return 'From the future!';
 80:         }
 81:     }
 82:     elseif ($interval == 0) {
 83:         return 'Just now';
 84:     }
 85: 
 86:     foreach (array_reverse($t) as $name => $val) {
 87:         if ($interval_time[$name] = ($interval/$val > 0) ? floor($interval/$val) : false) {
 88:             $interval -= $val*$interval_time[$name];
 89:         }
 90:     }
 91: 
 92:     $interval_time = array_slice(array_filter($interval_time, function($v) { return $v > 0; }), 0, $precision);
 93: 
 94:     foreach($interval_time as $name => $val) {
 95:         if ($val > 1) {
 96:             $res[] = sprintf('%d %ss', $val, $name);
 97:         }
 98:         else {
 99:             $res[] = sprintf('%d %s', $val, $name);
100:         }
101:     }
102:     return implode(', ', $res);
103: }

599:     $domain = false;

602:         return false;
603:     }
604:     if (strpos($val, '@') !== false) {
605:         $local = substr($val, 0, strrpos($val, '@'));
