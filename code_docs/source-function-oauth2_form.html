183:     static private $services = array();
184:     static private $oauth2 = array();
185: 
186:     static public function add($id, $details) {
187:         self::$services[$id] = $details;
188:     }
189:     static public function oauth2_setup($config) {
190:         $settings = array();
191:         $ini_file = rtrim($config->get('app_data_dir', ''), '/').'/oauth2.ini';
192:         if (is_readable($ini_file)) {
193:             $settings = parse_ini_file($ini_file, true);
194:             if (!empty($settings)) {
195:                 foreach ($settings as $service => $vals) {
196:                     self::$services[$service]['client_id'] = $vals['client_id'];
197:                     self::$services[$service]['client_secret'] = $vals['client_secret'];
198:                     self::$services[$service]['redirect_uri'] = $vals['client_uri'];
199:                 }
200:             }
201:         }
202:         self::$oauth2 = $settings;
203:     }
204: 
205:     /**
206:      * @todo uasort
207:      */
208:     static public function option_list($current, $mod) {
209:         $res = '';
210:         foreach(self::$services as $id => $details) {
211:             $res .= '<option value="'.$mod->html_safe($id).'"';
212:             if ($id == $current) {
213:                 $res .= ' selected="selected"';
214:             }
215:             $res .= '>'.$mod->trans($details['name']);
216:             $res .= '</option>';
217:         }
218:         return $res;
219:     }
220: 
221:     static public function exists($id) {
222:         return array_key_exists($id, self::$services);
223:     }
224: 
225:     static public function details($id) {
226:         if (array_key_exists($id, self::$services)) {
227:             return self::$services[$id];
228:         }
229:         return array();
