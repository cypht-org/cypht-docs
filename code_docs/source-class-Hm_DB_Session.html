105:     /**
106:      * Build HTTP header "fingerprint"
107:      * @param object $request request details
108:      * @return string fingerprint value
109:      */
110:     public function build_fingerprint($request, $input='') {
111:         $env = $request->server;
112:         $id = $input;
113:         $id .= (array_key_exists('REMOTE_ADDR', $env)) ? $env['REMOTE_ADDR'] : '';
114:         $id .= (array_key_exists('HTTP_USER_AGENT', $env)) ? $env['HTTP_USER_AGENT'] : '';
115:         $id .= (array_key_exists('REQUEST_SCHEME', $env)) ? $env['REQUEST_SCHEME'] : '';
116:         $id .= (array_key_exists('HTTP_ACCEPT_LANGUAGE', $env)) ? $env['HTTP_ACCEPT_LANGUAGE'] : '';
117:         $id .= (array_key_exists('HTTP_ACCEPT_CHARSET', $env)) ? $env['HTTP_ACCEPT_CHARSET'] : '';
118:         $id .= (array_key_exists('HTTP_HOST', $env)) ? $env['HTTP_HOST'] : '';
119:         return hash('sha256', $id);
120:     }

300:     public function start($request) {
301:         if (array_key_exists($this->cname, $request->cookie)) {
302:             session_id($request->cookie[$this->cname]);
303:         }
304:         list($secure, $path, $domain) = $this->set_session_params($request);

306:         Hm_Functions::session_start();

309:             session_regenerate_id(true);
