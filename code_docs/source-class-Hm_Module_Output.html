 10: /**
 11:  * Module data management. These functions provide an interface for modules (both handler and output)
 12:  * to fetch data set by other modules and to return their own output. Handler modules must use these
 13:  * methods to set a response, output modules must if the format is AJAX, otherwise they should return
 14:  * an HTML5 string
 15:  */ 
 16: trait Hm_Module_Output {
 17: 
 18:     /* module output */
 19:     protected $output = array();
 20: 
 21:     /* protected output keys */
 22:     protected $protected = array();
 23: 
 24:     /* list of appendable keys */
 25:     protected $appendable = array();
 26: 
 27:     /**
 28:      * Add a name value pair to the output array
 29:      * @param string $name name of value to store
 30:      * @param mixed $value value
 31:      * @param bool $protected true disallows overwriting
 32:      * @return bool true on success
 33:      */
 34:     public function out($name, $value, $protected=true) {
 35:         if (in_array($name, $this->protected, true)) {
 36:             Hm_Debug::add(sprintf('MODULES: Cannot overwrite protected %s with %s', $name, print_r($value,true)));
 37:             return false;
 38:         }
 39:         if (in_array($name, $this->appendable, true)) {
 40:             Hm_Debug::add(sprintf('MODULES: Cannot overwrite appendable %s with %s', $name, print_r($value,true)));
 41:             return false;
 42:         }
 43:         if ($protected) {
 44:             $this->protected[] = $name;
 45:         }
 46:         $this->output[$name] = $value;
 47:         return true;
 48:     }
 49: 
 50:     /**
 51:      * append a value to an array, create it if does not exist
 52:      * @param string $name array name
 53:      * @param string $value value to add
 54:      * @return bool true on success
 55:      */
 56:     public function append($name, $value) {
 57:         if (in_array($name, $this->protected, true)) {
 58:             Hm_Debug::add(sprintf('MODULES: Cannot overwrite %s with %s', $name, print_r($value,true)));
 59:             return false;
 60:         }
 61:         if (array_key_exists($name, $this->output)) {
 62:             if (is_array($this->output[$name])) {
 63:                 $this->output[$name][] = $value;
 64:                 return true;
 65:             }
 66:             else {
 67:                 Hm_Debug::add(sprintf('Tried to append %s to scaler %s', $value, $name));
 68:                 return false;
 69:             }
 70:         }
 71:         else {
 72:             $this->output[$name] = array($value);
 73:             $this->appendable[] = $name;
 74:             return true;
 75:         }
 76:     }
 77: 
 78:     /**
 79:      * Concatenate a value
 80:      * @param string $name name to add to
 81:      * @param string $value value to add
 82:      * @return bool true on success
 83:      */
 84:     public function concat($name, $value) {
 85:         if (array_key_exists($name, $this->output)) {
 86:             if (is_string($this->output[$name])) {
 87:                 $this->output[$name] .= $value;
 88:                 return true;
 89:             }
 90:             else {
 91:                 Hm_Debug::add('Could not append %s to %s', print_r($value,true), $name);
 92:                 return false;
 93:             }
 94:         }
 95:         else {
 96:             $this->output[$name] = $value;
 97:             return true;
 98:         }
 99:     }

117:     /**
118:      * Fetch an output value
119:      * @param string $name key to fetch the value for
120:      * @param mixed $default default return value if not found
121:      * @param string $typed if a default value is given, typecast the result to it's type
122:      * @return mixed value if found or default

124:     public function get($name, $default=NULL, $typed=true) {
125:         if (array_key_exists($name, $this->output)) {
126:             $val = $this->output[$name];
127:             if (!is_null($default) && $typed) {

130:                     settype($val, gettype($default));
131:                 }
132:             }

134:         }

161: /**
162:  * Base class for data input processing modules, called "handler modules"
163:  *
164:  * All modules that deal with processing input data extend from this class.
165:  * It provides access to input and state through the following member variables:
166:  *
167:  * $session      The session interface object
168:  * $request      The HTTP request details object
169:  * $config       The site config object
170:  * $user_config  The user settings object for the current user
171:  *
172:  * Modules that extend this class need to override the process function
173:  * Modules can pass information to the output modules using the out() and append() methods,
174:  * and see data from other modules with the get() method
175:  * @abstract
176:  */

178: 
179:     use Hm_Module_Output;

182:     public $session = false;
183: 
184:     /* request object */
185:     public $request = false;
186: 
187:     /* site configuration object */
188:     public $config = false;
189: 
190:     /* current request id */
191:     protected $page = false;
192: 
193:     /* user settings */
194:     protected $user_config = false;

196:     /**
197:      * Assign input and state sources
198:      * @param object $parent instance of the Hm_Request_Handler class
199:      * @param bool $logged_in true if currently logged in
200:      * @param array $output data from handler modules
201:      * @param array $protected list of protected output names
202:      * @param string $page page id
203:      * @return void
204:      */
205:     public function __construct($parent, $logged_in, $output=array(), $protected=array(), $page) {
206:         $this->session = $parent->session;
207:         $this->request = $parent->request;
208:         $this->page = $page;
209:         $this->config = $parent->site_config;

215:     /**
216:      * Validate a form key. If this is a non-empty POST form from an
217:      * HTTP request or AJAX update, it will take the user to the home
218:      * page if the page_key value is either not present or not valid
219:      * @return void
220:      */
221:     public function process_key() {
222: 
223:         Hm_Request_Key::load($this->session, $this->request);
224: 
225:         if (empty($this->request->post)) {
226:             return false;
227:         }
228: 
229:         $key = array_key_exists('hm_page_key', $this->request->post) ? $this->request->post['hm_page_key'] : false;
230:         $valid = Hm_Request_Key::validate($key);
231:         if (!$valid) {
232:             if ($this->request->type == 'AJAX') {
233:                 if (DEBUG_MODE) {
234:                     Hm_Debug::add('REQUEST KEY check failed');
235:                     Hm_Debug::load_page_stats();
236:                     Hm_Debug::show('log');
237:                 }
238:                 Hm_Functions::cease(json_encode(array('status' => 'not callable')));;
239:                 return 'exit';
240:             }
241:             else {
242:                 if ($this->session->loaded) {
243:                     $this->session->destroy($this->request);
244:                 }
245:                 Hm_Debug::add('REQUEST KEY check failed');
246:                 Hm_Dispatch::page_redirect('?page=home');
247:                 return 'redirect';
248:             }
249:         }
250:         return false;
251:     }

253:     /**
254:      * Process an HTTP POST form
255:      * @param array $form list of required field names in the form
256:      * @return array tuple with a bool indicating success, and an array of valid form values
257:      */
258:     public function process_form($form) {
259:         $post = $this->request->post;
260:         $success = false;
261:         $new_form = array();
262:         foreach($form as $name) {
263:             if (array_key_exists($name, $post) && (trim($post[$name]) || (($post[$name] === '0' ||  $post[$name] === 0 )))) {
264:                 $new_form[$name] = $post[$name];
265:             }
266:         }
267:         if (count($form) == count($new_form)) {
268:             $success = true;
269:         }
270:         return array($success, $new_form);
271:     }

279: /**
280:  * Base class for output modules
281:  * All modules that output data to a request must extend this class and define
282:  * an output() method. It provides form validation, html sanitizing,
283:  * and string translation services to modules
284:  * @abstract
285:  */

288:     use Hm_Module_Output;
289:

291:     protected $lstr = array();
292: 
293:     /* langauge name */
294:     protected $lang = false;
295: 
296:     /* UI layout direction */
297:     protected $dir = 'ltr';
298: 
299:     /* Output format (AJAX or HTML5) */
300:     protected $format = false;
301: 
302:     /**
303:      * Constructor
304:      * @param array $input data from handler modules
305:      * @param array $protected list of protected keys
306:      * @return void
307:      */
308:     function __construct($input, $protected) {
309:         $this->output = $input;
310:         $this->protected = $protected;
311:     }
312: 
313:     /**
314:      * Return a translated string if possible
315:      * @param the $string string to be translated
316:      * @return string translated string
317:      */
318:     public function trans($string) {
319:         if (array_key_exists($string, $this->lstr)) {
320:             if ($this->lstr[$string] === false) {
321:                 return $this->html_safe($string);
322:             }
323:             else {
324:                 return $this->html_safe($this->lstr[$string]);
325:             }
326:         }
327:         else {
328:             Hm_Debug::add(sprintf('TRANSLATION NOT FOUND :%s:', $string));
329:         }
330:         return $this->html_safe($string);
331:     }
332: 
333:     /**
334:      * Build output by calling module specific output functions
335:      * @param string $formt output type, either HTML5 or AJAX
336:      * @param array $lang_str list of language translation strings
337:      * @return mixed module output, a string for HTML5 format,
338:      *               and an array for AJAX
339:      */
340:     public function output_content($format, $lang_str, $protected) {
341:         $this->lstr = $lang_str;
342:         $this->format = str_replace('Hm_Format_', '', $format);
343:         if (array_key_exists('interface_lang', $lang_str)) {
344:             $this->lang = $lang_str['interface_lang'];
345:         }
346:         if (array_key_exists('interface_direction', $lang_str)) {
347:             $this->dir = $lang_str['interface_direction'];
348:         }
349:         return $this->output($format);
350:     }

355:      * @return string sanitized value
356:      */
357:     public function html_safe($string) {
358:         return htmlspecialchars($string, ENT_QUOTES | ENT_HTML5, 'UTF-8');
359:     }

365:      */
366:     abstract protected function output();
