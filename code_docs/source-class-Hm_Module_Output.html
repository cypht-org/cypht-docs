592:     /**
593:      * Add a module to a single page
594:      *
595:      * @param string $page the page to assign the module to
596:      * @param string $module the module to add
597:      * @param bool $logged_in true if the module requires the user to be logged in
598:      * @param string $marker the module to insert before or after
599:      * @param string $placement "before" or "after" the $marker module
600:      * @param bool $queue true to attempt to re-insert the module later on failure
601:      * @param string $source the module set containing this module
602:      *
603:      * @return void
604:      */
605:     public static function add($page, $module, $logged_in, $marker=false, $placement='after', $queue=true, $source=false) {
606:         $inserted = false;
607:         if (!array_key_exists($page, self::$module_list)) {
608:             self::$module_list[$page] = array();
609:         }
610:         if (array_key_exists($page, self::$module_list) && array_key_exists($module, self::$module_list[$page])) {
611:             Hm_Debug::add(sprintf("Already registered module re-attempted: %s", $module));
612:             return;
613:         }
614:         if (!$source) {
615:             $source = self::$source;
616:         }
617:         if ($marker) {
618:             $mods = array_keys(self::$module_list[$page]);
619:             $index = array_search($marker, $mods);
620:             if ($index !== false) {
621:                 if ($placement == 'after') {
622:                     $index++;
623:                 }
624:                 $list = self::$module_list[$page];
625:                 self::$module_list[$page] = array_merge(array_slice($list, 0, $index), 
626:                     array($module => array($source, $logged_in)),
627:                     array_slice($list, $index));
628:                 $inserted = true;
629:             }
630:         }
631:         else {
632:             $inserted = true;
633:             self::$module_list[$page][$module] = array($source, $logged_in);
634:         }
635:         if (!$inserted) {
636:             if ($queue) {
637:                 Hm_Debug::add(sprintf('queueing module %s', $module));
638:                 self::$module_queue[] = array($page, $module, $logged_in, $marker, $placement);
639:             }
640:             else {
641:                 Hm_Debug::add(sprintf('failed to insert module %s on %s', $module, $page));
642:             }
643:         }
644:     }

675:      * @param array $modules list of modules

680:         $keys = array_keys($modules);
