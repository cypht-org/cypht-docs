 96:     public function get_path($username) {
 97:         $path = $this->site_config->get('user_settings_dir', false);
 98:         return sprintf('%s/%s.txt', $path, $username);

101:     /**
102:      * Load the settings for a user
103:      *
104:      * @param string $username username
105:      * @param string $key key to decrypt the user data
106:      *
107:      * @return void
108:      */
109:     public function load($username, $key) {
110:         $source = $this->get_path($username);
111:         if (is_readable($source)) {
112:             $str_data = file_get_contents($source);
113:             if ($str_data) {
114:                 $data = @unserialize(Hm_Crypt::plaintext($str_data, $key));
115:                 if (is_array($data)) {
116:                     $this->config = array_merge($this->config, $data);
117:                     $this->set_tz();
118:                 }
119:             }
120:         }
121:     }

124:      * Reload from outside input
125:      *
126:      * @param array $data new user data
127:      *
128:      * @return void

132:         $this->set_tz();

172:     /**
173:      * Load the user settings from the DB
174:      *
175:      * @param string $username username
176:      * @param string $key encryption key
177:      *
178:      * @return void
179:      */
180:     public function load($username, $key) {
181:         if ($this->connect()) {
182:             $sql = $this->dbh->prepare("select * from hm_user_settings where username=?");
183:             if ($sql->execute(array($username))) {
184:                 $data = $sql->fetch();
185:                 if (!$data || !array_key_exists('settings', $data)) {
186:                     $sql = $this->dbh->prepare("insert into hm_user_settings values(?,?)");
187:                     if ($sql->execute(array($username, ''))) {
188:                         Hm_Debug::add(sprintf("created new row in hm_user_settings for %s", $username));
189:                         $this->config = array();
190:                     }
191:                 }
192:                 else {
193:                     $data = @unserialize(Hm_Crypt::plaintext($data['settings'], $key));
194:                     if (is_array($data)) {
195:                         $this->config = array_merge($this->config, $data);
196:                         $this->set_tz();
197:                     }
198:                 }
199:             }
200:         }
201:     }

204:      * Reload from outside input

212:         $this->set_tz();

217:      *
218:      * @return bool true on success
219:      */

229:      * Save user settings to the DB
230:      *

232:      * @param string $key encryption key
233:      *
234:      * @return void
235:      */
236:     public function save($username, $key) {
237:         $config = Hm_Crypt::ciphertext(serialize($this->config), $key);
238:         if ($this->connect()) {
239:             $sql = $this->dbh->prepare("update hm_user_settings set settings=? where username=?");
240:             if ($sql->execute(array($config, $username)) && $sql->rowCount() == 1) {
241:                 Hm_Debug::add(sprintf("Saved user data to DB for %s", $username));
242:                 return true;
243:             }
244:         }
245:         return false;
