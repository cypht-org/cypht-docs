113:  * Output the current configuration setup
114:  * @subpackage developer/output
115:  */
116: class Hm_Output_config_map extends Hm_Output_Module {
117:     /**
118:      * Show pages, module assignments, and input filters
119:      */
120:     protected function output() {
121:     $res = '<div class="content_title">'.$this->trans('Configuration Map').'</div><table class="config_map">';
122:     $handlers = array();
123:     $outputs = array();
124:     $server_info = $this->get('server_info', array());
125:     if (!empty($server_info)) {
126:         $handlers = $server_info['handlers'];
127:         ksort($handlers);
128:         $outputs = $server_info['output'];
129:     }
130:     foreach ($handlers as $page => $mods) {
131:         $res .= '<tr><td colspan="2" class="config_map_page" data-target="c'.$page.'">'.$page.'</td></tr>';
132:         $res .= '<tr><th class="c'.$page.'" >Handler Modules</th><th class="c'.$page.'" >Source</th></tr>';
133:         foreach ($mods as $name => $vals) {
134:             $res .= '<tr><td class="hmod c'.$page.'">'.$name.'</td><td class="hmod_val c'.$page.'">'.$vals[0].'</td></tr>';
135:         }
136:         if (array_key_exists($page, $outputs)) {
137:             $res .= '<tr><th class="c'.$page.'" >Output Modules</th><th class="c'.$page.'" >Source</th></tr>';
138:             foreach($outputs[$page] as $name => $vals) {
139:                 $res .= '<tr><td class="omod c'.$page.'">'.$name.'</td><td class="omod_val c'.$page.'">'.$vals[0].'</td></tr>';
140:             }
141:         }
142:     }
143:     $res .= '</table>';
144:     return $res;
145:     }
146: }
