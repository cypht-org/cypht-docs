 14: class Hm_Handler_swipe_2fa_check extends Hm_Handler_Module {
 15:     public function process() {
 16: 
 17:         /* new session or one not passed the second auth */
 18:         if ($this->session->loaded || $this->session->get('2fa_required', false)) {
 19: 
 20:             /* ini file location */
 21:             $ini_file = rtrim($this->config->get('app_data_dir', ''), '/').'/swipeidentity.ini';
 22: 
 23:             /* data for the swipe api */
 24:             $swipe_username = $this->session->get('username', false);
 25:             $swipe_address = $this->request->server['REMOTE_ADDR'];
 26:             $required = true;
 27: 
 28:             /* get api config and object */
 29:             list($api, $api_config) = setup_swipe_api($ini_file);
 30:             $started = start_api($api, $api_config);
 31:             if (!$started) {
 32:                 $this->out('2fa_fatal', true);
 33:             }
 34: 
 35:             /* get current 2fa state */
 36:             if (!array_key_exists('2fa_sms_response', $this->request->post)) {
 37:                 $state = get_secondfactor_state($api, $api_config, $swipe_username, $swipe_address);
 38:             }
 39:             else {
 40:                 $state = RC_SMS_DELIVERED;
 41:             }
 42: 
 43:             /* pass a key and no redirect flag to the output modules */
 44:             $this->out('no_redirect', true);
 45:             $this->out('2fa_key', Hm_Request_Key::generate());
 46: 
 47:             $sms_number = false;
 48:             $sms_response = false;
 49: 
 50:             /* if the user has not registered a phone number yet look for one in POST */
 51:             if ($state == NEED_REGISTER_SMS && array_key_exists('sms_number', $this->request->post)) {
 52: 
 53:                 /* remove non numeric delimiters */
 54:                 $sms_number = preg_replace("/[^\d]/", "", $this->request->post['sms_number']);
 55: 
 56:                 /* US phone numbers only for now */
 57:                 if (preg_match("/^1\d{10}$/", $sms_number)) {
 58:                     $submit_number = $sms_number;
 59: 
 60:                     /* set the phone number using the api */
 61:                     $api->setUserSmsNumber($swipe_username, $api_config["com.swipeidentity.api.appcode"], $submit_number);
 62: 
 63:                     /* refecth the status */
 64:                     $state = get_secondfactor_state($api, $api_config, $swipe_username, $swipe_address);
 65: 
 66:                     /* number rejected by swipe */
 67:                     if ($state == NEED_REGISTER_SMS) {
 68:                         $this->out('2fa_error', 'Invalid phone number');
 69:                     }
 70:                 }
 71:                 else {
 72:                     $this->out('2fa_error', 'Invalid phone number format');
 73:                 }
 74: 
 75:             }
 76:             /* the sms was sent, look for a sms code in POST */
 77:             elseif ($state == RC_SMS_DELIVERED && array_key_exists('2fa_sms_response', $this->request->post)) {
 78: 
 79:                 if (preg_match("/^\d{5}$/", $this->request->post['2fa_sms_response'])) {
 80: 
 81:                     $sms_response = $this->request->post['2fa_sms_response'];
 82: 
 83:                     /* validate the sms response with the api */
 84:                     $resp = $api->answerSMS($swipe_username, $api_config["com.swipeidentity.api.appcode"], $sms_response);
 85: 
 86:                     /* success! allow the user to login */
 87:                     if ($resp->getReturnCode() == RC_SMS_ANSWER_ACCEPTED) {
 88:                         $required = false;
 89:                     }
 90:                     else {
 91:                         $state = get_secondfactor_state($api, $api_config, $swipe_username, $swipe_address);
 92:                         $this->out('2fa_error', 'Response did not match! A new sms code has been sent');
 93:                     }
 94:                 }
 95:                 else {
 96:                     $this->out('2fa_error', 'Incorrectly formatted response, please re-enter the sms code');
 97:                 }
 98:             }
 99: 
100:             /* if required is true we still have not completed the 2fa */
101:             if ($required) {
102: 
103:                 /* pass required flag to modules */
104:                 $this->session->set('2fa_required', true);
105:                 $this->out('2fa_required', true);
106:                 $this->out('2fa_state', $state);
107: 
108:                 /* close the session early */
109:                 $this->session->close_early();
110:             }
111:             else {
112: 
113:                 /* unset any previously set required flags */
114:                 $this->session->set('2fa_required', false);
115:                 $this->out('2fa_required', false);
116:             }
117:         }
118:     }
119: }
