  5:  * @package framework
  6:  * @subpackage modules
  7:  */

  9: 
 10: /**
 11:  * Trait used as the basic logic for module management
 12:  *
 13:  * Two classes use this trait, Hm_Handler_Modules and Hm_Output_Modules.
 14:  * These are the interfaces module sets use (indirectly) to interact with a request
 15:  * and product output to the browser.
 16:  */
 17: trait Hm_Modules {
 18: 
 19:     /* holds the module to page assignment list */
 20:     private static $module_list = array();
 21: 
 22:     /* current module set name, used for error tracking and limiting php file inclusion */
 23:     private static $source = false;
 24: 
 25:     /* a retry queue for modules that fail to insert immediately */
 26:     private static $module_queue = array();
 27: 
 28:     /* queue for delayed module insertion for all pages */
 29:     private static $all_page_queue = array();
 30: 
 31:     /**
 32:      * Queue a module to be added to all defined pages
 33:      * @param string $module the module to add
 34:      * @param bool $logged_in true if the module requires the user to be logged in
 35:      * @param string $marker the module to insert before or after
 36:      * @param string $placement "before" or "after" the $marker module
 37:      * @param string $source the module set containing this module
 38:      * return void
 39:      */
 40:     public static function queue_module_for_all_pages($module, $logged_in, $marker, $placement, $source) {
 41:         self::$all_page_queue[] = array($module, $logged_in, $marker, $placement, $source);
 42:     }
 43: 
 44:     /**
 45:      * Process queued modules and add them to all pages
 46:      * @return void
 47:      */
 48:     public static function process_all_page_queue() {
 49:         foreach (self::$all_page_queue as $mod) {
 50:             self::add_to_all_pages($mod[0], $mod[1], $mod[2], $mod[3], $mod[4]);
 51:         }
 52:     }
 53: 
 54:     /**
 55:      * Load a complete formatted module list
 56:      * @param array $mod_list list of module assignments
 57:      * @return void
 58:      */
 59:     public static function load($mod_list) {
 60:         self::$module_list = $mod_list;
 61:     }
 62: 
 63:     /**
 64:      * Assign the module set name
 65:      * @param string $source the name of the module set (imap, pop3, core, etc)
 66:      * @return void
 67:      */
 68:     public static function set_source($source) {
 69:         self::$source = $source;
 70:     }
 71: 
 72:     /**
 73:      * Add a module to every defined page
 74:      * @param string $module the module to add
 75:      * @param bool $logged_in true if the module requires the user to be logged in
 76:      * @param string $marker the module to insert before or after
 77:      * @param string $placement "before" or "after" the $marker module
 78:      * @param string $source the module set containing this module
 79:      * @return void
 80:      */
 81:     public static function add_to_all_pages($module, $logged_in, $marker, $placement, $source) {
 82:         foreach (self::$module_list as $page => $modules) {
 83:             if (!preg_match("/^ajax_/", $page)) {
 84:                 self::add($page, $module, $logged_in, $marker, $placement, true, $source);
 85:             }
 86:         }
 87:     }
 88: 
 89:     /**
 90:      * Add a module to a single page
 91:      * @param string $page the page to assign the module to
 92:      * @param string $module the module to add
 93:      * @param bool $logged_in true if the module requires the user to be logged in
 94:      * @param string $marker the module to insert before or after
 95:      * @param string $placement "before" or "after" the $marker module
 96:      * @param bool $queue true to attempt to re-insert the module later on failure
 97:      * @param string $source the module set containing this module
 98:      * @return void
 99:      */
100:     public static function add($page, $module, $logged_in, $marker=false, $placement='after', $queue=true, $source=false) {
101:         $inserted = false;
102:         if (!array_key_exists($page, self::$module_list)) {
103:             self::$module_list[$page] = array();
104:         }
105:         if (array_key_exists($page, self::$module_list) && array_key_exists($module, self::$module_list[$page])) {
106:             Hm_Debug::add(sprintf("Already registered module re-attempted: %s", $module));
107:             return;
108:         }
109:         if (!$source) {
110:             $source = self::$source;
111:         }
112:         if ($marker) {
113:             $mods = array_keys(self::$module_list[$page]);
114:             $index = array_search($marker, $mods);
115:             if ($index !== false) {
116:                 if ($placement == 'after') {
117:                     $index++;
118:                 }
119:                 $list = self::$module_list[$page];
120:                 self::$module_list[$page] = array_merge(array_slice($list, 0, $index), 
121:                     array($module => array($source, $logged_in)),
122:                     array_slice($list, $index));
123:                 $inserted = true;
124:             }
125:         }
126:         else {
127:             $inserted = true;
128:             self::$module_list[$page][$module] = array($source, $logged_in);
129:         }
130:         if (!$inserted) {
131:             if ($queue) {
132:                 self::$module_queue[] = array($page, $module, $logged_in, $marker, $placement, $source);
133:             }
134:             else {
135:                 Hm_Debug::add(sprintf('failed to insert module %s on %s', $module, $page));
136:             }
137:         }
138:     }
139: 
140:     /**
141:      * Replace an already assigned module with a different one
