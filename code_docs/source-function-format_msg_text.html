 10:  * Build meta information for a message list

 16: function message_list_meta($input, $output_mod) {
 17:     if (!array_key_exists('list_meta', $input) || !$input['list_meta']) {
 18:         return '';
 19:     }
 20:     $limit = 0;
 21:     $since = false;
 22:     $times = array(
 23:         'today' => 'Today',
 24:         '-1 week' => 'Last 7 days',
 25:         '-2 weeks' => 'Last 2 weeks',
 26:         '-4 weeks' => 'Last 4 weeks',
 27:         '-6 weeks' => 'Last 6 weeks',
 28:         '-6 months' => 'Last 6 months',
 29:         '-1 year' => 'Last year'
 30:     );
 31:     if (array_key_exists('per_source_limit', $input)) {
 32:         $limit = $input['per_source_limit'];
 33:     }
 34:     if (!$limit) {
 35:         $limit = DEFAULT_PER_SOURCE;
 36:     }
 37:     if (array_key_exists('message_list_since', $input)) {
 38:         $since = $input['message_list_since'];
 39:     }
 40:     if (!$since) {
 41:         $since = DEFAULT_SINCE;
 42:     }
 43:     $dt = sprintf('%s', strtolower($output_mod->trans($times[$since])));
 44:     $max = sprintf($output_mod->trans('sources@%d each'), $limit);
 45: 
 46:     return '<div class="list_meta">'.
 47:         $output_mod->html_safe($dt).
 48:         '<b>-</b>'.
 49:         '<span class="src_count"></span> '.$max.
 50:         '<b>-</b>'.

 54: /**
 55:  * Build a human readable interval string
 56:  * @subpackage core/functions
 57:  * @param string $date_str date string parsable by strtotime()
 58:  * @return string
 59:  */
 60: function human_readable_interval($date_str) {
 61:     $precision     = 2;
 62:     $interval_time = array();
 63:     $date          = strtotime($date_str);
 64:     $interval      = time() - $date;
 65:     $res           = array();
 66: 
 67:     $t['second'] = 1;
 68:     $t['minute'] = $t['second']*60;
 69:     $t['hour']   = $t['minute']*60;
 70:     $t['day']    = $t['hour']*24;
 71:     $t['week']   = $t['day']*7;
 72:     $t['month']  = $t['day']*30;
 73:     $t['year']   = $t['week']*52;
 74: 
 75:     if ($interval < 0) {
 76:         return 'From the future!';
 77:     }
 78:     elseif ($interval == 0) {
 79:         return 'Just now';
 80:     }
 81:     foreach (array_reverse($t) as $name => $val) {
 82:         if ($interval_time[$name] = ($interval/$val > 0) ? floor($interval/$val) : false) {
 83:             $interval -= $val*$interval_time[$name];
 84:         }
 85:     }
 86:     $interval_time = array_slice(array_filter($interval_time, function($v) { return $v > 0; }), 0, $precision);
 87:     foreach($interval_time as $name => $val) {
 88:         if ($val > 1) {
 89:             $res[] = sprintf('%d %ss', $val, $name);
 90:         }
 91:         else {
 92:             $res[] = sprintf('%d %s', $val, $name);
 93:         }
 94:     }
 95:     return implode(', ', $res);
 96: }

586: /**
587:  * Validate an E-mail using RFC 3696
588:  * @subpackage core/functions
589:  * @param string $val value to check
590:  * @param bool $allow_local flag to allow local addresses with no domain
591:  * @return bool
592:  */
593: function is_email($val, $allow_local=false) {
594:     $domain = false;
595:     $local = false;
596:     if (!trim($val) || strlen($val) > 320) {
597:         return false;
598:     }
599:     if (strpos($val, '@') !== false) {
600:         $local = substr($val, 0, strrpos($val, '@'));
601:         $domain = substr($val, (strrpos($val, '@') + 1));
602:     }
603:     else {
604:         $local = $val;
605:     }
606:     if (!$local || (!$allow_local && !$domain)) {
607:         return false;
608:     }
609:     else {
610:         if ($domain && !validate_domain_full($domain)) {
611:             return false;
612:         }
613:         if (!validate_local_full($local)) {
614:             return false;
615:         }
616:     }
617:     return true;

626: function validate_domain_full($val) {
627:     /* check for a dot, max allowed length and standard ASCII characters */
628:     if (strpos($val, '.') === false || strlen($val) > 255 || preg_match("/[^A-Z0-9\-\.]/i", $val) ||

637:  * @subpackage core/functions
638:  * @param string $val value to check
639:  * @return bool
640:  */
641: function validate_local_full($val) {
642:     /* check length, "." rules, and for characters > ASCII 127 */
643:     if (strlen($val) > 64 || $val{0} == '.' || $val{(strlen($val) -1)} == '.' || strstr($val, '..') ||
644:         preg_match('/[^\x00-\x7F]/',$val)) {
645:         return false;
646:     }
647:     /* remove escaped characters and quoted strings */

649:     $local = preg_replace("/\"[^\"]+\"/", '', $local);

653:         return true;
654:     }
655:     return false;
