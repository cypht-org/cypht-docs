600        $this->out('is_mobile', $this->request->mobile);

607            if (!$this->session->is_active()) {

609            }

612                $this->out('redirect_url', $this->config->get('redirect_after_login'));

616            $this->session->check($this->request);
617        }

619            $this->out('changed_settings', $this->session->get('changed_settings', array()), false);
620            $this->out('username', $this->session->get('username'));
621        }
622        if ($this->validate_request) {
623            Hm_Request_Key::load($this->session, $this->request, $this->session->loaded);
624            $this->validate_method($this->session, $this->request);
625            $this->process_key();
626            if (!$this->config->get('disable_origin_check', false)) {
627                $this->validate_origin($this->session, $this->request, $this->config);
628            }
629        }
630    }
